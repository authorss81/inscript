  space_shooter.ins 
// A complete space shooter — no engine setup required
scene Main {
    let player = Entity { pos: Vec2(400, 500), health: 100, speed: 180.0 }

    on_update(dt: float) {
        player.pos += Input.wasd() * player.speed * dt

        if Input.just_pressed(Key.Space) {
            spawn(Bullet { pos: player.pos, vel: Vec2(0, -600) })
        }
    }

    on_draw {
        Draw.sprite("ship.png", player.pos)
        Draw.hud_text("HP: " + player.health, Vec2(10, 10))
    }
}
Everything a game needs
Built into the language. No external libraries required.
🧱

Entity-Component System

Archetype-based ECS with automatic batching, component queries, and system scheduling built in.

⚡

2D + 3D Physics

Full rigid-body dynamics, GJK/EPA collision, joints, character controller, and vehicle physics.

✨

Shader System

Write shaders in InScript syntax. Transpiles to GLSL, WGSL, or HLSL. Hot-reload while running.

🌐

Networking & Multiplayer

TCP/UDP/WebSocket, sync vars, RPC, lobby rooms, lag compensation, and TURN relay out of the box.

🤖

Game AI

FSM, behavior trees, A* navmesh pathfinding, Boids flocking, influence maps, GOAP planning.

🖼️

Declarative UI

Flexbox-inspired layout, 15 widget types, anchor system for HUDs, theming, animations.

🚀

Bytecode VM

Register-based VM with 60+ opcodes, constant pool, closures, GC, profiler, and .insb binary format.

🌍

Web Export

One command builds a PWA-ready HTML5 bundle with Canvas2D, Web Audio, WebSocket, and IndexedDB.

🔒

Type Safety

Hindley-Milner style type inference, generics, optional types, and exhaustive pattern matching.

Install
Three ways to get InScript running.
pip
npm
Docker
# Python 3.10+ required
pip install inscript-lang

# Verify
inscript --version    # InScript 0.5.0

# Run a game
inscript mygame.ins

# Start REPL
inscript --repl

# Build for web
inscript build --target web mygame.ins

  Structs & Methods
struct Bullet {
    pos: Vec2
    vel: Vec2
    dmg: int = 10

    fn update(dt: float) {
        self.pos += self.vel * dt
    }

    fn is_offscreen() -> bool {
        return self.pos.y < 0
    }
}

  AI State Machine
ai Enemy {
    state Idle {
        on_update(dt) {
            if can_see(player) {
                switch_to(Chase)
            }
        }
    }
    state Chase {
        on_update(dt) {
            navigate_to(player.pos)
        }
    }
}
