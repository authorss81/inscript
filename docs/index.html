InScript

A game-first scripting language. scene, on_update, ai, and shader are first-class keywords.

scene GameScene {
    let score: int = 0

    on_update(dt: float) {
        if input.key_pressed("space") {
            score += 1
        }
    }

    on_draw {
        draw.text(10, 10, f"Score: {score}", WHITE)
    }
}
